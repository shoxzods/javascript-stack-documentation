// @keyframes:
css:

@theme {
  --animate-myAnimation: changeColor 1s ease-in infinite;
}

@keyframes changeColor {
  from {
    color:red;
  }

  to {
    color: green;
  }
}

js:

theme: {
    extend: {
      keyframes: {
         changeColor: {
            from:{
                color:"red"
            },

            to: {
              color:"green"
            }
         }
      },

      animation: {
        changeColor: "changeColor 1s infinite ease-in",
      }
    },
  },


/* html */
/* 
  <p class="animate-myAnimation">lorem</p>
*/
